CREATE TABLE Category (
    ID NUMBER(10) PRIMARY KEY,
    Name VARCHAR2(25),
    Last_Update TIMESTAMP
);


CREATE TABLE Language (
    ID NUMBER(10) PRIMARY KEY,
    Name VARCHAR2(20),
    Last_Update TIMESTAMP
);

CREATE TABLE Film (
    ID NUMBER(10) PRIMARY KEY,
    LanguageID NUMBER(10),
    Title VARCHAR2(255),
    Description VARCHAR2(255),
    Release_Year NUMBER(4),
    Rental_Duration NUMBER(10),
    Rental_Rate NUMBER(19, 0),
    Length NUMBER(2),
    Replacement_Cost NUMBER(19, 0),
    Rating NUMBER(10),
    Last_Update TIMESTAMP,
    Special_Features VARCHAR2(255),
    Fulltext VARCHAR2(255),
    CONSTRAINT fk_language FOREIGN KEY (LanguageID) REFERENCES Language(ID)
);

CREATE TABLE Actor (
    ID NUMBER(10) PRIMARY KEY,
    First_Name VARCHAR2(255),
    Last_Name VARCHAR2(255),
    Last_Update TIMESTAMP
);

CREATE TABLE Film_Actor (
    FilmID NUMBER(10),
    ActorID NUMBER(10),
    Last_Update TIMESTAMP,
    FOREIGN KEY (FilmID) REFERENCES Film(ID),
    FOREIGN KEY (ActorID) REFERENCES Actor(ID)
);

CREATE TABLE Film_Category (
    FilmID NUMBER(10),
    CategoryID NUMBER(10),
    Last_Update TIMESTAMP,
    FOREIGN KEY (FilmID) REFERENCES Film(ID),
    FOREIGN KEY (CategoryID) REFERENCES Category(ID)
);

CREATE TABLE Inventory (
    ID NUMBER(10),
    FilmID NUMBER(10),
    Last_Update TIMESTAMP,
    PRIMARY KEY (ID),
    FOREIGN KEY (FilmID) REFERENCES Film(ID)
);


CREATE TABLE Country (
    ID NUMBER(10),
    Country VARCHAR2(50),
    Last_Update TIMESTAMP,
    PRIMARY KEY (ID)
);

CREATE TABLE City (
    ID NUMBER(10),
    CountryID NUMBER(10),
    City VARCHAR2(50),
    Last_Update TIMESTAMP,
    PRIMARY KEY (ID),
    FOREIGN KEY (CountryID) REFERENCES Country(ID)
);

CREATE TABLE Address (
    ID NUMBER(10),
    CityID NUMBER(10),
    Address VARCHAR2(50),
    Address2 VARCHAR2(50),
    District NUMBER(20),
    Postal_Code VARCHAR2(10),
    Phone VARCHAR2(20),
    Last_Update TIMESTAMP,
    PRIMARY KEY (ID),
    FOREIGN KEY (CityID) REFERENCES City(ID)
);

CREATE TABLE Store (
    ID NUMBER(10),
    AddressID NUMBER(10),
    Last_Update TIMESTAMP,
    PRIMARY KEY (ID),
    FOREIGN KEY (AddressID) REFERENCES Address(ID)
);


CREATE TABLE Customer (
    ID NUMBER(10),
    AddressID NUMBER(10),
    AddressColumn NUMBER(10) NOT NULL,
    First_Name VARCHAR2(255),
    Last_Name VARCHAR2(255),
    Email VARCHAR2(50),
    Active CHAR(1),
    Create_Date TIMESTAMP,
    Last_Update TIMESTAMP,
    PRIMARY KEY (ID),
    FOREIGN KEY (AddressID) REFERENCES Address(ID)
);

CREATE TABLE Staff (
    ID NUMBER(10),
    AddressID NUMBER(10),
    StoreID NUMBER(10),
    First_Name VARCHAR2(255),
    Last_Name VARCHAR2(255),
    Email VARCHAR2(50),
    Active CHAR(1),
    Username VARCHAR2(16),
    Password VARCHAR2(40),
    Last_Update TIMESTAMP,
    PictureURL VARCHAR2(80),
    PRIMARY KEY (ID),
    FOREIGN KEY (AddressID) REFERENCES Address(ID),
    FOREIGN KEY (StoreID) REFERENCES Store(ID)
);

CREATE TABLE Rental (
    ID NUMBER(10),
    StaffID NUMBER(10),
    CustomerID NUMBER(10),
    InventoryID NUMBER(10),
    Rental_Date TIMESTAMP,
    Return_Date TIMESTAMP,
    Last_Update TIMESTAMP,
    PRIMARY KEY (ID),
    FOREIGN KEY (StaffID) REFERENCES Staff(ID),
    FOREIGN KEY (CustomerID) REFERENCES Customer(ID),
    FOREIGN KEY (InventoryID) REFERENCES Inventory(ID)
);

CREATE TABLE Payment (
    ID NUMBER(10),
    RentalID NUMBER(10),
    CustomerID NUMBER(10),
    StaffID NUMBER(10),
    Amount NUMBER(19, 0),
    Payment_Date TIMESTAMP,
    PRIMARY KEY (ID),
    FOREIGN KEY (RentalID) REFERENCES Rental(ID), 
    FOREIGN KEY (CustomerID) REFERENCES Customer(ID),
    FOREIGN KEY (StaffID) REFERENCES Staff(ID)
);
